func maxProfit(prices []int) int {
  l := 0
  r := 1

  maxProfit := 0
  maxIndex := len(prices) - 1

  for {
    if r > maxIndex {
      break
    }

    profit := prices[r] - prices[l]
    if profit < 0 {
      l = r
    } else {
      maxProfit = max(maxProfit, profit)
    }
    r += 1
  }
  return maxProfit
}

func max(a, b int) int {
  if a > b {
      return a
  }
  return b
}

  maxProfit = 0

    if profit >= 0:

  return maxProfit
